Python - Financial Risk Analysis

import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
import matplotlib.pyplot as plt

# Load quarterly financial data
df = pd.read_csv('q4_2024_data.csv')

# Calculate risk metrics
def calculate_var(returns, confidence_level=0.95):
    """Calculate Value at Risk"""
    return np.percentile(returns, (1 - confidence_level) * 100)

# Risk assessment model
risk_model = RandomForestClassifier(n_estimators=100, random_state=42)
risk_scores = risk_model.fit_predict(df[['loan_amount', 'credit_score', 'income']])

print(f"Portfolio VaR (95%): ${calculate_var(df['returns']) * 1e6:.2f}M")

Database Performance Optimization

SQL - Customer Segmentation Query

WITH customer_metrics AS (
    SELECT 
        customer_id,
        SUM(transaction_amount) as total_volume,
        COUNT(*) as transaction_count,
        AVG(account_balance) as avg_balance,
        CASE 
            WHEN SUM(transaction_amount) > 100000 THEN 'Premium'
            WHEN SUM(transaction_amount) > 50000 THEN 'Gold' 
            ELSE 'Standard'
        END as customer_tier
    FROM transactions t
    JOIN accounts a ON t.account_id = a.account_id
    WHERE transaction_date >= '2024-10-01'
    GROUP BY customer_id
)
SELECT customer_tier, COUNT(*) as customer_count,
       AVG(total_volume) as avg_volume
FROM customer_metrics 
GROUP BY customer_tier;
